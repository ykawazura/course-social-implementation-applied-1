\documentclass[tikz, border=8pt]{standalone}
\usepackage{tikz}
\usepackage{xcolor}
\usepackage{amsmath}
\usepackage{luatexja}
\usetikzlibrary{arrows.meta, decorations.pathreplacing, calc}

\begin{document}
\begin{tikzpicture}[
  dot/.style={circle, fill, inner sep=2pt},
  wall/.style={very thick},
  gridline/.style={thin, gray!50},
  >=Latex
]

% x 列位置: 0, 1, i-1, i, i+1, Nx
% (1 と i-1 の間・i+1 と Nx の間を広めに取り「...」を挿入)
\def\xA{0}    % 0
\def\xB{1.2}  % 1
\def\xC{2.8}  % i-1
\def\xD{4.0}  % i
\def\xE{5.2}  % i+1
\def\xF{6.8}  % Nx

% y 行位置: 0, 1, j-1, j, j+1, Ny
\def\yA{0}    % 0
\def\yB{1.2}  % 1
\def\yC{2.8}  % j-1
\def\yD{4.0}  % j
\def\yE{5.2}  % j+1
\def\yF{6.8}  % Ny

% --- グリッド線（縦）---
\foreach \xi in {\xA, \xB, \xC, \xD, \xE, \xF}{
  \draw[gridline] (\xi, \yA) -- (\xi, \yF);
}

% --- グリッド線（横）---
\foreach \yj in {\yA, \yB, \yC, \yD, \yE, \yF}{
  \draw[gridline] (\xA, \yj) -- (\xF, \yj);
}

% --- 格子点 ---
\foreach \xi in {\xA, \xB, \xC, \xD, \xE, \xF}{
  \foreach \yj in {\yA, \yB, \yC, \yD, \yE, \yF}{
    \node[dot, fill=black!70] at (\xi, \yj) {};
  }
}

% --- 壁 ---
\draw[wall, red!70!black]  (\xA, \yA) -- (\xF, \yA);
\draw[wall, blue!70!black] (\xA, \yF) -- (\xF, \yF);
% 上壁ラベルは後で下壁の左端に揃えるため、ここでは座標だけ定義
\coordinate (topwallpos) at (0, {6.8+0.35});

% --- x 方向インデックスラベル ---
\node[below=4pt, font=\footnotesize] at (\xA, \yA) {$0$};
\node[below=4pt, font=\footnotesize] at (\xB, \yA) {$1$};
\node[below=2pt, font=\small]        at ({(\xB+\xC)/2}, \yA) {$\cdots$};
\node[below=4pt, font=\footnotesize] at (\xC, \yA) {$i{-}1$};
\node[below=4pt, font=\footnotesize] at (\xD, \yA) {$i$};
\node[below=4pt, font=\footnotesize] at (\xE, \yA) {$i{+}1$};
\node[below=2pt, font=\small]        at ({(\xE+\xF)/2}, \yA) {$\cdots$};
\node[below=4pt, font=\footnotesize] at (\xF, \yA) {$N_x$};

% --- dx ブレース ---
\draw[decorate, decoration={brace, amplitude=5pt, mirror}, thick]
  (\xC, -0.7) -- (\xD, -0.7)
  node[midway, below=5pt, font=\small] {$\Delta x$};

% --- 下壁ラベル ---
% 下壁ラベル（名前付きノード）
\node[font=\small\bfseries, red!70!black, anchor=west] (botwall)
  at (5.0, -0.85) {下壁（高温 $T=1$）};

% 上壁ラベル（左端を下壁の左端に揃える）
\node[font=\small\bfseries, blue!70!black, anchor=west]
  at (botwall.west |- topwallpos) {上壁（低温 $T=0$）};

% --- y 方向インデックスラベル（最左列●の直左）---
\node[font=\footnotesize, anchor=east] at (-0.35, \yA) {$0$};
\node[font=\footnotesize, anchor=east] at (-0.35, \yB) {$1$};
\node[font=\small,        anchor=east] at (-0.35, {(\yB+\yC)/2}) {$\vdots$};
\node[font=\footnotesize, anchor=east] at (-0.35, \yC) {$j{-}1$};
\node[font=\footnotesize, anchor=east] at (-0.35, \yD) {$j$};
\node[font=\footnotesize, anchor=east] at (-0.35, \yE) {$j{+}1$};
\node[font=\small,        anchor=east] at (-0.35, {(\yE+\yF)/2}) {$\vdots$};
\node[font=\footnotesize, anchor=east] at (-0.35, \yF) {$N_y$};

% --- dy ブレース ---
\draw[decorate, decoration={brace, amplitude=5pt}, thick]
  (-1.3, \yC) -- (-1.3, \yD)
  node[midway, left=5pt, font=\small] {$\Delta y$};

% --- 着目点 (i, j) ---
\node[dot, fill=red!80, inner sep=3.5pt] at (\xD, \yD) {};
\node[above right=2pt, font=\small, red!80!black] at (\xD, \yD) {$(i,\,j)$};

% --- 5点ステンシル矢印 ---
\draw[->, dashed, red!60, thick] (\xD, \yD) -- (\xC, \yD);
\draw[->, dashed, red!60, thick] (\xD, \yD) -- (\xE, \yD);
\draw[->, dashed, red!60, thick] (\xD, \yD) -- (\xD, \yC);
\draw[->, dashed, red!60, thick] (\xD, \yD) -- (\xD, \yE);

\end{tikzpicture}
\end{document}
